Lambda expressions

Java (lambdas) 
OptionalInt sum =
  IntStream.of(1, 2, 3, 4, 5)
    // 1, 3, 5
  .filter(i -> (i % 2) != 0)
  // 2, 4, 6
  .map((i) -> { i++; return i; })
  // 2 + 4 + 6
  .reduce((i, a) -> a + i);
ES6 (arrows) 
let sum =
  Array.of(1, 2, 3, 4, 5)
  // 1, 3, 5
  .filter(i => (i % 2) === 0)
  // 2, 4, 6
  .map((i) => { i++; return i; })
  // 2 + 4 + 6
  .reduce((i, a) => i + a);

Yeeeeah! Java don't give up

Varargs

Java (varargs) 
public int f(int x, int... y) {
  // y is int[]
	return x * y.length;
}
f(3, 5, 7) == 6
ES6 (rest and spread) 
function f(x, ...y) {
  // y is an Array
  return x * y.length;
}
f(3, 5, 7) == 6

function f(x, y, z) {
  return x + y + z;
}
// Pass each elem of array as argument
f(...[1,2,3]) == 6

Modules

In ES6 file is named module. It may import or export functions, objects or primitives from or to other modules

// lib/math.js
export default function sum(x, y) {
  return x + y;
}
export const pi = 3.141593;

// example1.js
import * as math from "lib/math";
console.log("2π = " + math.sum(math.pi, math.pi));

// example2.js
import sum, {pi} from "lib/math";
console.log("2π = " + sum(pi, pi));
